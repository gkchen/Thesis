\section{\ce{Be+ + H2O} All Thermochemically Allowed Channels}

\begin{align}

	\dot{\ce{HOC}}(t) & = - k_{\ref{r: X+HOC->XH}} \rho_{\ce{X}} \ce{HOC}(t) \\

\end{align}
